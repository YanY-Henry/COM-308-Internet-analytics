from math import sqrt

def error(r_rdd, test):
    # calculate squared error for each user in test RDD
    joined_rdd = test.map(lambda x: ((x['userId'], x['movieId']), x['rating'])) \
        .join(r_rdd.map(lambda x: ((x[0], x[1]), x[2])))

    squared_errors = joined_rdd \
        .map(lambda x: ((x[0][0]), ((x[1][0] - x[1][1]) ** 2, 1))) \
        .reduceByKey(lambda x, y: (x[0] + y[0], x[1] + y[1]))

    # calculate mean squared error for each user
    rmse_values = squared_errors.map(lambda x: sqrt(x[1][0] / x[1][1]))

    # calculate overall error
    total_error = rmse_values.sum()

    # calculate the number of distinct users in the test RDD
    num_users = test.map(lambda x: x['userId']).distinct().count()

    # calculate error by dividing total_error by the number of distinct users
    err = total_error / num_users

    return err

from math import sqrt

def error(r_rdd, test):
    # Calculate squared error for each user in test RDD
    joined_rdd = test.map(lambda x: ((x['userId'], x['movieId']), x['rating'])) \
        .join(r_rdd.map(lambda x: ((x[0], x[1]), x[2])))

    squared_errors = joined_rdd \
        .map(lambda x: ((x[0][0]), ((x[1][0] - x[1][1]) ** 2, 1))) \
        .reduceByKey(lambda x, y: (x[0] + y[0], x[1] + y[1]))

    # Calculate mean squared error for each user
    rmse_values = squared_errors.map(lambda x: sqrt(x[1][0] / x[1][1]))

    # Calculate overall error
    total_error = rmse_values.sum()

    # Calculate the number of distinct users in the test RDD
    num_users = test.map(lambda x: x['userId']).distinct().count()

    # Calculate error by dividing total_error by the number of distinct users
    err = total_error / num_users

    return err

Here we choose rank = 10. The trade-off between choosing a lower and a higher rank:

Low Rank Models: Simple and fast, but may not capture complexities in the data due to fewer parameters, leading to higher bias.

High Rank Models: Can learn finer details and subtleties in the data, improving accuracy, but are prone to overfitting and increased computational cost.

lambdas = [2*1e-4, 5*1e-4, 1e-3, 5*1e-3, 0.01, 0.05, 0.1]

# Set a fixed rank value
rank = 15

# Store the error for each lambda
errors = []

for lambda_ in lambdas:
    # Train the model using ALS
    model = ALS.train(ratings, rank=rank, lambda_=lambda_)

    # Generate predictions and transform format to (user, item, prediction)
    predictions = model.predictAll(validation).map(lambda x: ((x[0], x[1]), x[2]))
    transformed_predictions = predictions.map(lambda x: (x[0][0], x[0][1], x[1]))

    # Compute the error for this model
    current_error = error(transformed_predictions, test)
    errors.append((lambda_, current_error))
    print(f'Lambda: {lambda_}, Error: {current_error}')

# Find the optimal lambda with the lowest error
best_lambda, best_error = min(errors, key=lambda x: x[1])

Here is the observation based on current knowledge:

Based on the observation, the x-axis probability represents the quality of the movie. In the graph, movies that lean in the negative direction of the x-axis (i.e., the left side of the graph) generally have higher Rotten Tomatoes ratings; while movies with lower ratings are generally distributed in the positive direction of the x-axis (i.e., the right side of the graph). While this pattern is not absolute, it is generally indicative of the meaning of the x-axis. y-axis is most likely related to the genre of the movie. Most of the movies distributed in the positive direction of the y-axis (i.e., the upper part of the graph) contain thriller or horror elements; and the further to the negative direction of the y-axis (i.e., the lower part of the graph), the higher the probability that the movie is a comedy. This means that the more relaxing a movie is, the more likely it is to be on the lower side of the graph.

For cluster of movies, notice that there is a small clustering between the movies City of God and Ichi the Killer, both of which contain violent elements. Another cluster appears between Independence Day, Princess Diaries, and Spider-Man, probably because all three movies have some elements of advanture. Still one more cluster appears between Schindler's List, The King's Speech and Accused, which have in common that they are all categorized as drama. Finally, there is also a small cluster between Forrest Gump and Control, probably because both of these movies are kind of biographical in nature and focus on the lives of the main characters.
