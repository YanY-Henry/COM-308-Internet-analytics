import numpy as np

def estimate_power_law_alpha(degree_sequence):
    # Filter out degrees less than a chosen xmin.
    # This value is often chosen as the point where the power-law behavior seems to start
    xmin = min(degree_sequence)
    degrees = np.array([degree for degree in degree_sequence if degree >= xmin])
    
    n = len(degrees)  # number of observations
    # Calculate the summation term in the MLE formula
    summation = np.sum(np.log(degrees / xmin))
    
    # Estimate alpha using the MLE approximation
    alpha = 1 + n * (summation ** -1)
    
    return alpha, xmin

alpha, xmin = estimate_power_law_alpha(degree_sequence)

print(f"Estimated power-law alpha: {alpha}")
print(f"Chosen xmin: {xmin}")

def average_reachable_nodes(G, max_r, num_samples=1000):
    reachable_nodes = []
    # Sample a subset of nodes if the graph is large
    nodes = list(G.nodes())
    sampled_nodes = random.sample(nodes, num_samples)
        
    for r in range(1, max_r + 1):
        reachable_count = []
        for node in sampled_nodes:
            reachable = nx.single_source_shortest_path_length(G, node, cutoff=r)
            reachable_count.append(len(reachable) - 1)  # subtract 1 to exclude the node itself
        avg_reachable = np.mean(reachable_count)
        reachable_nodes.append(avg_reachable)
    return reachable_nodes

from scipy.optimize import curve_fit
from scipy.special import gamma

# Define the probability density function for the exponentiated exponential distribution
def expexp_pdf(x, lamb, beta):
    return lamb * beta * np.exp(-lamb * x**beta) * x**(beta - 1)

Exercise 2.6: Network identification¶

Network1 exhibits characteristics of a small-world network, such as a short average path length and rapid increase in node reachability, which is likely to be a subgraph of Internet routers. Internet router networks are designed to handle large volumes of data transmission, hence they are structured to allow information to be passed quickly between routers, which explains the short average path lengths.

Network2 has a very long average shortest path length and slower growth in node reachability, which is more consistent with the characteristics of a city road network. In urban road networks, due to geographical and planning constraints, paths are often not the most direct, and the number of roads (edges) connected at each intersection (node) is limited.

Therefore, based on the provided data and analysis, we can infer that Network1 is likely a subgraph of Internet routers, while Network2 is likely the road network around New York City.
